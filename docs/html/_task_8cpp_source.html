71void CTask::Backtrack(const size_t m){
72 if(m == 0)m_nCount++; else{
73 for(size_t i=0; i<m; i++){
74 const size_t j = m - 1; //first element to swap in permutation
75 const size_t k = j - i; //second element to swap in permutation
76 const size_t dx = m_nPerm[k] + j; //diagonal index
77 const size_t bx = m_nPerm[k] - m + m_nSize; //back-diagonal index
78
79 if(m_bBackDiagonal[bx] && m_bDiagonal[dx]){ //diagonal & back-diagonal unused
80 std::swap(m_nPerm[j], m_nPerm[k]); //permute
81
82 m_bBackDiagonal[bx] = m_bDiagonal[dx] = false; //mark back-diagonal and diagonal used
83 Backtrack(j); //recurse on smaller array
84 m_bBackDiagonal[bx] = m_bDiagonal[dx] = true; //mark back-diagonal and diagonal unused
85
86 std::swap(m_nPerm[j], m_nPerm[k]); //unpermute
87 } //if
88 } //for
89 } //else
90} //Backtrack
